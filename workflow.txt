package com.example.redis.auth;

import java.time.Instant;

public record CachedToken(
        String token,
        Instant expiresAt,
        String accessKeyId   // fingerprint of the AWS creds that signed this token
) {
    /** True if we should proactively refresh (2-minute buffer before hard expiry). */
    public boolean shouldRefresh() {
        return Instant.now().isAfter(expiresAt.minusSeconds(120));
    }

    /** True if the creds that made this token are no longer active. */
    public boolean credentialsChanged(String currentAccessKeyId) {
        return !this.accessKeyId.equals(currentAccessKeyId);
    }
}
package com.example.redis.auth;

import io.lettuce.core.RedisCredentials;
import io.lettuce.core.RedisCredentialsProvider;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import reactor.core.publisher.Mono;
import software.amazon.awssdk.auth.credentials.AwsCredentials;
import software.amazon.awssdk.auth.credentials.AwsCredentialsProvider;
import software.amazon.awssdk.auth.credentials.DefaultCredentialsProvider;
import software.amazon.awssdk.regions.Region;
import software.amazon.awssdk.services.elasticache.auth.ElastiCacheAuthTokenGenerator;
import software.amazon.awssdk.services.elasticache.auth.GenerateAuthTokenRequest;

import java.time.Duration;
import java.time.Instant;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.ConcurrentHashMap;

/**
 * Lettuce CredentialsProvider that:
 *  1. Caches IAM auth tokens keyed by region:host:port
 *  2. Invalidates the cache when IRSA rotates AWS credentials (AKID fingerprint check)
 *  3. Proactively refreshes 2 minutes before token expiry (12-min effective TTL vs 14-min hard limit)
 *  4. Deduplicates concurrent refresh calls with a CompletableFuture mutex
 *  5. Falls back to the stale token if STS is momentarily unreachable
 */
public class RedisIamTokenProvider implements RedisCredentialsProvider {

    private static final Logger log = LoggerFactory.getLogger(RedisIamTokenProvider.class);

    private static final Duration TOKEN_LIFETIME    = Duration.ofMinutes(14);
    private static final Duration STALE_GRACE       = Duration.ofMinutes(5);

    private final String host;
    private final int    port;
    private final String region;
    private final String iamUser;     // the Redis IAM user configured on the cluster
    private final AwsCredentialsProvider credentialsProvider;
    private final ElastiCacheAuthTokenGenerator tokenGenerator;

    // cache key → CachedToken
    private final ConcurrentHashMap<String, CachedToken> tokenCache   = new ConcurrentHashMap<>();
    // cache key → in-flight refresh future  (acts as a per-key mutex)
    private final ConcurrentHashMap<String, CompletableFuture<String>> inflight = new ConcurrentHashMap<>();

    public RedisIamTokenProvider(String host, int port, String region, String iamUser) {
        this.host               = host;
        this.port               = port;
        this.region             = region;
        this.iamUser            = iamUser;
        this.credentialsProvider = DefaultCredentialsProvider.create();
        this.tokenGenerator     = ElastiCacheAuthTokenGenerator.builder()
                                        .region(Region.of(region))
                                        .credentialsProvider(credentialsProvider)
                                        .build();
    }

    /** Constructor for testing — inject your own providers. */
    public RedisIamTokenProvider(
            String host, int port, String region, String iamUser,
            AwsCredentialsProvider credentialsProvider,
            ElastiCacheAuthTokenGenerator tokenGenerator) {
        this.host                = host;
        this.port                = port;
        this.region              = region;
        this.iamUser             = iamUser;
        this.credentialsProvider = credentialsProvider;
        this.tokenGenerator      = tokenGenerator;
    }

    // ── Lettuce entry point ───────────────────────────────────────────────────

    @Override
    public Mono<RedisCredentials> resolveCredentials() {
        return Mono.fromCompletionStage(this::resolveAsync);
    }

    // ── Core logic ────────────────────────────────────────────────────────────

    private CompletableFuture<RedisCredentials> resolveAsync() {
        return getTokenAsync()
                .thenApply(token -> RedisCredentials.just(iamUser, token));
    }

    public CompletableFuture<String> getTokenAsync() {
        String cacheKey       = buildCacheKey();
        AwsCredentials creds  = credentialsProvider.resolveCredentials();
        String currentAkid    = creds.accessKeyId();

        CachedToken cached = tokenCache.get(cacheKey);

        boolean needsRefresh =
                cached == null
                || cached.credentialsChanged(currentAkid)   // ← THE FIX: IRSA rotated
                || cached.shouldRefresh();                   // proactive 12-min refresh

        if (!needsRefresh) {
            return CompletableFuture.completedFuture(cached.token());
        }

        // ── Mutex: if a refresh is already in flight, reuse it ────────────────
        return inflight.computeIfAbsent(cacheKey, k -> {
            log.debug("Redis IAM: refreshing token [key={}, reason={}]",
                    cacheKey, refreshReason(cached, currentAkid));

            return generateTokenAsync(cacheKey, currentAkid)
                    .exceptionally(ex -> handleRefreshFailure(ex, cached, cacheKey))
                    .whenComplete((token, ex) -> inflight.remove(cacheKey));
        });
    }

    // ── Token generation ──────────────────────────────────────────────────────

    private CompletableFuture<String> generateTokenAsync(String cacheKey, String akid) {
        return CompletableFuture.supplyAsync(() -> {
            GenerateAuthTokenRequest request = GenerateAuthTokenRequest.builder()
                    .hostname(host)
                    .port(port)
                    .userName(iamUser)
                    .build();

            String token    = tokenGenerator.generateAuthToken(request);
            Instant expires = Instant.now().plus(TOKEN_LIFETIME);

            tokenCache.put(cacheKey, new CachedToken(token, expires, akid));
            log.debug("Redis IAM: token refreshed [akid={}, expiresAt={}]", akid, expires);
            return token;
        });
    }

    // ── Stale-on-error fallback ───────────────────────────────────────────────

    private String handleRefreshFailure(Throwable ex, CachedToken stale, String cacheKey) {
        if (stale != null && Instant.now().isBefore(stale.expiresAt().plus(STALE_GRACE))) {
            log.warn("Redis IAM: token refresh failed, using stale token for up to {}s [key={}]",
                    STALE_GRACE.toSeconds(), cacheKey, ex);
            return stale.token();
        }
        throw new RuntimeException("Redis IAM: token refresh failed and no usable stale token", ex);
    }

    // ── Helpers ───────────────────────────────────────────────────────────────

    private String buildCacheKey() {
        return region + ":" + host + ":" + port;
    }

    private String refreshReason(CachedToken cached, String currentAkid) {
        if (cached == null)                          return "no-cache";
        if (cached.credentialsChanged(currentAkid)) return "credential-rotation";
        return "proactive-ttl";
    }
}
package com.example.redis.auth;

import io.lettuce.core.RedisClient;
import io.lettuce.core.RedisURI;
import io.lettuce.core.api.StatefulRedisConnection;
import io.lettuce.core.api.sync.RedisCommands;

import java.time.Duration;

public class RedisClientFactory {

    public static RedisCommands<String, String> create(
            String host, int port, String region, String iamUser) {

        RedisIamTokenProvider provider = new RedisIamTokenProvider(host, port, region, iamUser);

        RedisURI uri = RedisURI.builder()
                .withHost(host)
                .withPort(port)
                .withSsl(true)                 // IAM auth requires TLS
                .withVerifyPeer(true)
                .withTimeout(Duration.ofSeconds(5))
                .withAuthentication(provider)  // Lettuce calls resolveCredentials() on each connect/reconnect
                .build();

        RedisClient client = RedisClient.create(uri);
        StatefulRedisConnection<String, String> connection = client.connect();
        return connection.sync();
    }
